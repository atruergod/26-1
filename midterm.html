항목 평가 이유
베이지안 IRT로 일차원성 검증 ✅ 타당 2PL/1PL 모델 적합 후 사후분포로 판단 가능
Stan(CmdStanPy) 사용 ✅ 타당 IRT 모델링에 매우 적합한 도구
모의 데이터 생성 ✅ 타당 논문 구조를 재현하기 위한 합리적 접근
DIF를 베이지안으로 구현 ✅ 타당 사후분포 비교로 불확실성 정량화 가능

⚠️ 주의할 차이점¶

원논문(SIBTEST) 베이지안 대안
빈도주의 검정 (p < .05) 사후분포의 credible interval
β 계수 + 표준오차 β의 사후분포 전체
단일 추정값 불확실성 포함한 완전한 분포
일차원성: 주성분분석(PCA) 일차원성: 2PL 모델 적합도 + 잔차 구조 분석

💡 접근 전략¶

원논문 SIBTEST → 베이지안 대응
─────────────────────────────────────────
1단계: 일차원성 검증
  PCA (분산 20% 기준) → 베이지안 2PL IRT 적합 + 
                        factor loading 사후분포 검토

2단계: DIF 도출
  SIBTEST β 계수    → 집단별 문항 난이도 차이의 사후분포
  p < .05 기준       → 95% credible interval이 0 포함 여부

import pandas as pd
import numpy as np

def generate_irt_responses(n_persons, n_items, ability_mean, gender_label, dif_items=None, seed=42):
    np.random.seed(seed)
    # 능력치 생성
    theta = np.random.normal(ability_mean, 1.0, n_persons)
    # 문항 모수 생성 (a: 변별도, b: 난이도)
    a = np.random.lognormal(0, 0.2, n_items)
    b = np.random.normal(0, 1.0, n_items)
    
    # DIF 적용 (연구집단인 여성에게만 적용)
    if dif_items and gender_label == 1:
        for idx, shift in dif_items.items():
            b[idx] += shift # 난이도 변화
            
    # 응답 생성 (2PL 로지스틱 모델)
    responses = np.zeros((n_persons, n_items))
    for i in range(n_persons):
        prob = 1 / (1 + np.exp(-a * (theta[i] - b)))
        responses[i] = np.random.binomial(1, prob)
        
    return responses, theta, a, b

        m_res, _, _, _ = generate_irt_responses(counts['male'], N_ITEMS, 0.0, 0, seed=r_num)
        # 여성(연구)
        f_res, _, _, _ = generate_irt_responses(counts['female'], N_ITEMS, 0.05, 1, 
                                               dif_items=dif_items_vocab if domain == '어휘문법' else None, 
                                               seed=r_num+1)
        
        # DataFrame 구축 (생략된 로직 보강)
        for i, res in enumerate(np.vstack([m_res, f_res])):
            row = {'domain': domain, 'gender': 0 if i < counts['male'] else 1}
            for j in range(N_ITEMS):
                row[f'item_{j+1:02d}'] = res[j]
            all_data_list.append(row)

# Cronbach's Alpha 계산 함수
def cronbach_alpha(data_matrix):
    """
    Cronbach's Alpha 계산
    α = (k / (k-1)) * (1 - Σσ²_i / σ²_total)
    """
    k = data_matrix.shape[1]
    item_variances = data_matrix.var(axis=0, ddof=1)
    total_variance = data_matrix.sum(axis=1).var(ddof=1)
    alpha = (k / (k - 1)) * (1 - item_variances.sum() / total_variance)
    return alpha

# 셀 4: Stan 모델 파일 작성 - 베이지안 2PL IRT
# ============================================================
"""
베이지안 2PL IRT 모델:
- theta_i ~ Normal(0, 1)          : 잠재 능력
- b_j ~ Normal(0, 1)              : 문항 난이도
- a_j ~ LogNormal(0, 0.5)         : 문항 변별도 (양수 제약)
- y_ij ~ Bernoulli(logit^{-1}(a_j*(theta_i - b_j)))

일차원성 검증:
- 모델 적합 후 문항 변별도의 사후분포 검토
- Q3 통계량 (문항 잔차 상관) 계산
"""

stan_2pl_code = """
// ================================================
// 베이지안 2PL IRT 모델 (일차원성 검증용)
// ================================================
data {
    int<lower=1> N;           // 응시자 수
    int<lower=1> J;           // 문항 수
    array[N, J] int<lower=0, upper=1> Y;  // 응답 행렬
}

parameters {
    // 능력 모수: 표준정규분포 사전분포
    vector[N] theta;          
    
    // 문항 난이도
    vector[J] b;             
    
    // 문항 변별도 (로그 스케일로 양수 보장)
    vector[J] log_a;         
}

transformed parameters {
    // 변별도 (양수)
    vector<lower=0>[J] a = exp(log_a);
}

model {
    // ── 사전분포 (Priors) ──
    // 능력 모수: 식별성을 위해 표준정규
    theta ~ normal(0, 1);
    
    // 문항 난이도: 약한 정보적 사전분포
    b ~ normal(0, 1);
    
    // 변별도: LogNormal (양수, 대략 0.5~2.0 범위)
    log_a ~ normal(0, 0.5);  // exp(0±0.5) ≈ 0.6~1.6
    
    // ── 우도 (Likelihood) ──
    for (i in 1:N) {
        for (j in 1:J) {
            real eta = a[j] * (theta[i] - b[j]);
            Y[i, j] ~ bernoulli_logit(eta);
        }
    }
}

generated quantities {
    // ── 사후 예측 분포 (Posterior Predictive) ──
    array[N, J] int<lower=0, upper=1> Y_rep;
    
    // ── 로그 우도 (모델 비교용) ──
    matrix[N, J] log_lik;
    
    for (i in 1:N) {
        for (j in 1:J) {
            real eta = a[j] * (theta[i] - b[j]);
            Y_rep[i, j] = bernoulli_logit_rng(eta);
            log_lik[i, j] = bernoulli_logit_lpmf(Y[i, j] | eta);
        }
    }
}
"""

# Stan 모델 파일 작성 - DIF 분석용
# ================================================
"""
베이지안 DIF 분석 모델:
- 집단(남/여)별 문항 난이도 차이를 직접 모델링
- delta_j = b_female_j - b_male_j : DIF 효과 크기
- delta_j = 0이면 DIF 없음
- 95% credible interval이 0을 포함하지 않으면 유의한 DIF

이 방식은 SIBTEST의 β와 유사한 해석 가능
"""

stan_dif_code = """
// ================================================
// 베이지안 DIF 분석 모델 (집단별 문항 난이도 차이)
// ================================================
data {
    int<lower=1> N_ref;       // 참조집단(남) 응시자 수
    int<lower=1> N_foc;       // 연구집단(여) 응시자 수
    int<lower=1> J;           // 문항 수
    
    array[N_ref, J] int<lower=0, upper=1> Y_ref;  // 참조집단 응답
    array[N_foc, J] int<lower=0, upper=1> Y_foc;  // 연구집단 응답
}

parameters {
    // ── 능력 모수 ──
    vector[N_ref] theta_ref;  // 참조집단 능력
    vector[N_foc] theta_foc;  // 연구집단 능력
    
    // ── 공통 문항 모수 ──
    vector[J] b;              // 공통 난이도 (기준선)
    vector[J] log_a;          // 공통 변별도 (로그 스케일)
    
    // ── DIF 효과 ──
    // delta_j > 0: 남자에게 유리 (여자에게 더 어려움)
    // delta_j < 0: 여자에게 유리 (남자에게 더 어려움)
    vector[J] delta;          // DIF 효과 크기
    
    // ── 하이퍼파라미터 ──
    real mu_ref;              // 참조집단 평균 능력
    real mu_foc;              // 연구집단 평균 능력
    real<lower=0> sigma_ref;  // 참조집단 능력 표준편차
    real<lower=0> sigma_foc;  // 연구집단 능력 표준편차
}

transformed parameters {
    vector<lower=0>[J] a = exp(log_a);    // 변별도
    vector[J] b_ref = b;                   // 참조집단 난이도 = 공통
    vector[J] b_foc = b + delta;           // 연구집단 난이도 = 공통 + DIF
}

model {
    // ── 사전분포 ──
    // 능력 모수
    theta_ref ~ normal(mu_ref, sigma_ref);
    theta_foc ~ normal(mu_foc, sigma_foc);
    
    // 하이퍼파라미터
    mu_ref ~ normal(0, 1);
    mu_foc ~ normal(0, 1);
    sigma_ref ~ exponential(1);
    sigma_foc ~ exponential(1);
    
    // 문항 모수
    b ~ normal(0, 1);
    log_a ~ normal(0, 0.5);
    
    // DIF 사전분포: Horseshoe-like sparse prior
    // 대부분의 문항은 DIF 없음을 가정 (sparse)
    delta ~ normal(0, 0.3);   // 약한 사전분포: 작은 DIF 선호
    
    // ── 우도 ──
    // 참조집단 (남)
    for (i in 1:N_ref) {
        for (j in 1:J) {
            real eta = a[j] * (theta_ref[i] - b_ref[j]);
            Y_ref[i, j] ~ bernoulli_logit(eta);
        }
    }
    
    // 연구집단 (여)
    for (i in 1:N_foc) {
        for (j in 1:J) {
            real eta = a[j] * (theta_foc[i] - b_foc[j]);
            Y_foc[i, j] ~ bernoulli_logit(eta);
        }
    }
}

generated quantities {
    // DIF 크기 분류 (Roussos & Stout 기준을 베이지안으로 재해석)
    // |delta| < 0.059: A (무시)
    // 0.059 <= |delta| < 0.088: B (중간)
    // |delta| >= 0.088: C (큰)
    array[J] int dif_class;
    vector[J] prob_dif_A;    // P(|delta| < 0.059)
    vector[J] prob_dif_B;    // P(0.059 <= |delta| < 0.088)
    vector[J] prob_dif_C;    // P(|delta| >= 0.088)
    
    for (j in 1:J) {
        real abs_delta = fabs(delta[j]);
        if (abs_delta < 0.059) {
            dif_class[j] = 0;  // A: 무시
        } else if (abs_delta < 0.088) {
            dif_class[j] = 1;  // B: 중간
        } else {
            dif_class[j] = 2;  // C: 큰
        }
        
        // 사후 확률 계산
        prob_dif_A[j] = (abs_delta < 0.059) ? 1.0 : 0.0;
        prob_dif_B[j] = (abs_delta >= 0.059 && abs_delta < 0.088) ? 1.0 : 0.0;
        prob_dif_C[j] = (abs_delta >= 0.088) ? 1.0 : 0.0;
    }
    
    // 집단별 평균 능력 차이
    real ability_gap = mu_foc - mu_ref;
}
"""

    f.write(stan_dif_code)

# DIF 분류 (Roussos & Stout 기준)
def classify_dif(delta_val):
    """
    |delta| < 0.059: A (무시할 수준)
    0.059 <= |delta| < 0.088: B (중간 수준)
    |delta| >= 0.088: C (큰 수준)
    양수: 남자에게 유리
    음수: 여자에게 유리
    """
    abs_d = abs(delta_val)
    if abs_d < 0.059:
        return 'A (무시)'
    elif abs_d < 0.088:
        return 'B (중간)'
    else:
        return 'C (큰)'

# DIF 유의성 판단 (95% HDI가 0을 포함하지 않으면 유의)
def is_significant_dif(hdi_lower, hdi_upper):
    return not (hdi_lower <= 0 <= hdi_upper)

# 결과 테이블 생성

for j in range(N_ITEMS_PER_DOMAIN):
    is_sig = is_significant_dif(delta_hdi[j, 0], delta_hdi[j, 1])

        'delta_sd': round(delta_sd[j], 3),
        'HDI_lower': round(delta_hdi[j, 0], 3),

    # 데이터 추출
    male_data = df[(df['domain'] == domain) & (df['gender'] == 0)][item_cols].values.astype(int)
    female_data = df[(df['domain'] == domain) & (df['gender'] == 1)][item_cols].values.astype(int)
    
    # 서브샘플
    n_m = min(300, male_data.shape[0])
    n_f = min(300, female_data.shape[0])
    
    np.random.seed(42)
    m_idx = np.random.choice(male_data.shape[0], n_m, replace=False)
    f_idx = np.random.choice(female_data.shape[0], n_f, replace=False)
    
    stan_data = {
        'N_ref': n_m,
        'N_foc': n_f,
        'J': N_ITEMS_PER_DOMAIN,
        'Y_ref': male_data[m_idx].tolist(),
        'Y_foc': female_data[f_idx].tolist()
    }
    
    print(f"  남: {n_m}명, 여: {n_f}명 → MCMC 샘플링...")
    
    # MCMC (간소화 버전 - 빠른 실행)
    fit = model_dif.sample(
        data=stan_data,
        chains=2,           # 빠른 실행을 위해 2체인
        iter_warmup=500,
        iter_sampling=1000,
        adapt_delta=0.88,
        seed=42,
        show_progress=False  # 출력 간소화
    )
    
    # 결과 추출
    idata = az.from_cmdstanpy(posterior=fit)
    delta_post = idata.posterior['delta'].values.reshape(-1, N_ITEMS_PER_DOMAIN)
    delta_m = delta_post.mean(axis=0)
    delta_s = delta_post.std(axis=0)
    delta_h = az.hdi(idata, var_names=['delta'])['delta'].values
    
    # 유의한 DIF 문항
    domain_results = []
    for j in range(N_ITEMS_PER_DOMAIN):
        is_sig = not (delta_h[j, 0] <= 0 <= delta_h[j, 1])
        domain_results.append({
            'domain': domain,
            'item': j + 1,
            'delta_mean': round(delta_m[j], 3),
            'delta_sd': round(delta_s[j], 3),
            'hdi_lower': round(delta_h[j, 0], 3),
            'hdi_upper': round(delta_h[j, 1], 3),
            'significant': is_sig,
            'direction': '남유리' if delta_m[j] > 0 else '여유리',
            'dif_class': classify_dif(delta_m[j]) if is_sig else 'A'
        })
    
    all_dif_results[domain] = domain_results
    
    sig_count = sum(r['significant'] for r in domain_results)
    male_fav = sum(r['significant'] and r['direction']=='남유리' for r in domain_results)
    female_fav = sum(r['significant'] and r['direction']=='여유리' for r in domain_results)
    
    print(f"  ✅ 완료: DIF 탐지 {sig_count}개 (남유리: {male_fav}, 여유리: {female_fav})")
